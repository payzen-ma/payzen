MODULE[01] anciennete {
  INPUTS  : salaire_base_26j, date_embauche, mois_paie
  OUTPUTS : anciennete_annees, taux_anciennete, prime_anciennete

  RULE anciennete.1 {
    anciennete_annees = DIFF_YEARS(date_embauche, LAST_DAY(mois_paie))
  }

  RULE anciennete.2 {
    WHEN anciennete_annees < 2    THEN taux_anciennete = 0.00
    WHEN anciennete_annees < 5    THEN taux_anciennete = 0.05
    WHEN anciennete_annees < 12   THEN taux_anciennete = 0.10
    WHEN anciennete_annees < 20   THEN taux_anciennete = 0.15
    WHEN anciennete_annees >= 20  THEN taux_anciennete = 0.20
    ;;
    ;; ATTENTION IA : la borne inférieure EST INCLUSE.
    ;; Exemple : anciennete_annees = 5 → taux = 10% (tranche [5, 12[)
    ;; Exemple : anciennete_annees = 2 → taux =  5% (tranche [2,  5[)
  }

  RULE anciennete.3 {
    prime_anciennete = ROUND(salaire_base_26j * taux_anciennete, 2)
    ;; BASE : salaire de base UNIQUEMENT — hors primes et indemnités
  }
}


MODULE[02] presence {
  INPUTS  : salaire_base_26j, jours_travailles, jours_feries, jours_conge
  OUTPUTS : jours_payes_total, salaire_base_mensuel

  RULE presence.1 {
    jours_payes_total = jours_travailles + jours_feries + jours_conge
  }

  RULE presence.2 {
    WHEN jours_payes_total >= 26
      THEN salaire_base_mensuel = salaire_base_26j
    WHEN jours_payes_total < 26
      THEN salaire_base_mensuel = ROUND(salaire_base_26j * jours_payes_total / 26, 2)
  }
}


MODULE[03] heures_supplementaires {
  INPUTS  : salaire_base_mensuel, prime_anciennete, heures_mois,
            h_sup_25pct, h_sup_50pct, h_sup_100pct
  OUTPUTS : taux_horaire, mont_hsupp_25, mont_hsupp_50, mont_hsupp_100, total_hsupp

  RULE hsupp.1 {
    base_hsupp   = salaire_base_mensuel + prime_anciennete
    taux_horaire = ROUND(base_hsupp / heures_mois, 4)
  }

  RULE hsupp.2 {
    mont_hsupp_25  = ROUND(h_sup_25pct  * taux_horaire * 1.25, 2)
    mont_hsupp_50  = ROUND(h_sup_50pct  * taux_horaire * 1.50, 2)
    mont_hsupp_100 = ROUND(h_sup_100pct * taux_horaire * 2.00, 2)
    total_hsupp    = mont_hsupp_25 + mont_hsupp_50 + mont_hsupp_100
  }
}


MODULE[04] indemnites_non_imposables {
  INPUTS  : salaire_base_26j, jours_travailles,
            ni_transport, ni_kilometrique, ni_tournee, ni_representation,
            ni_panier, ni_caisse, ni_salissure, ni_lait, ni_outillage,
            ni_aide_medicale, ni_gratif_sociale, ni_autres
  OUTPUTS : [pour chaque NI] ni_X_exo, ni_X_imposable
            total_ni_exonere, total_ni_excedent_imposable

  RULE ni.0 — Principe de scission {
    ;; Pour tout élément NI :
    ;;   partie_exoneree  = MIN(montant_saisi, plafond_applicable)
    ;;   partie_imposable = MAX(0, montant_saisi - plafond_applicable)
    ;; La partie_imposable rejoint le BRUT IMPOSABLE (MODULE 05)
  }

  RULE ni.1 — Transport {
    ni_transport_exo       = MIN(ni_transport, PLAFOND_NI_TRANSPORT)
    ni_transport_imposable = MAX(0, ni_transport - PLAFOND_NI_TRANSPORT)
  }

  RULE ni.2 — Kilométrique {
    ni_kilometrique_exo       = ni_kilometrique
    ni_kilometrique_imposable = 0
    ;; Pas de plafond mensuel fixe — taux 3 MAD/km pour trajets > 50 km
  }

  RULE ni.3 — Tournée {
    ni_tournee_exo       = MIN(ni_tournee, PLAFOND_NI_TOURNEE)
    ni_tournee_imposable = MAX(0, ni_tournee - PLAFOND_NI_TOURNEE)
  }

  RULE ni.4 — Représentation {
    PLAFOND_REP = salaire_base_26j * PLAFOND_NI_REPRESENTATION
    ni_representation_exo       = MIN(ni_representation, PLAFOND_REP)
    ni_representation_imposable = MAX(0, ni_representation - PLAFOND_REP)
    WARNING : "Vérifier éligibilité selon fonction (cadres supérieurs CNSS / PDG-DG DGI)"
  }

  RULE ni.5 — Panier {
    PLAFOND_PANIER = PLAFOND_NI_PANIER_JOUR * jours_travailles
    ni_panier_exo       = MIN(ni_panier, PLAFOND_PANIER)
    ni_panier_imposable = MAX(0, ni_panier - PLAFOND_PANIER)
  }

  RULE ni.6 — Caisse {
    ;; Divergence CNSS/DGI — utiliser DGI (le plus restrictif) pour IR
    ni_caisse_exo_cnss       = MIN(ni_caisse, PLAFOND_NI_CAISSE)
    ni_caisse_exo_dgi        = MIN(ni_caisse, PLAFOND_NI_CAISSE_DGI)
    ni_caisse_imposable_ir   = MAX(0, ni_caisse - PLAFOND_NI_CAISSE_DGI)
    ni_caisse_imposable_cnss = MAX(0, ni_caisse - PLAFOND_NI_CAISSE)
    AUDIT : "Divergence CNSS(239)/DGI(190) — IR basé sur DGI"
  }

  RULE ni.7 — Lait {
    ni_lait_exo_dgi        = MIN(ni_lait, PLAFOND_NI_LAIT_DGI)
    ni_lait_imposable_ir   = MAX(0, ni_lait - PLAFOND_NI_LAIT_DGI)
    ni_lait_imposable_cnss = MAX(0, ni_lait - PLAFOND_NI_LAIT)
    AUDIT : "Divergence CNSS(196)/DGI(150)"
  }

  RULE ni.8 — Outillage {
    ni_outillage_exo_dgi        = MIN(ni_outillage, PLAFOND_NI_OUTILLAGE_DGI)
    ni_outillage_imposable_ir   = MAX(0, ni_outillage - PLAFOND_NI_OUTILLAGE_DGI)
    ni_outillage_imposable_cnss = MAX(0, ni_outillage - PLAFOND_NI_OUTILLAGE)
    AUDIT : "Divergence CNSS(119)/DGI(100)"
  }

  RULE ni.9 — Salissure {
    ni_salissure_exo_dgi        = MIN(ni_salissure, PLAFOND_NI_SALISSURE_DGI)
    ni_salissure_imposable_ir   = MAX(0, ni_salissure - PLAFOND_NI_SALISSURE_DGI)
    ni_salissure_imposable_cnss = MAX(0, ni_salissure - PLAFOND_NI_SALISSURE)
    AUDIT : "Divergence CNSS(239)/DGI(210)"
  }

  RULE ni.10 — Aide médicale {
    ni_aide_medicale_exo       = ni_aide_medicale
    ni_aide_medicale_imposable = 0
    CONDITION : "Factures justificatives obligatoires"
  }

  RULE ni.11 — Gratifications sociales {
    ni_gratif_exo_cnss       = MIN(ni_gratif_sociale, PLAFOND_NI_GRATIF_ANNUEL)
    ni_gratif_exo_dgi        = MIN(ni_gratif_sociale, PLAFOND_NI_GRATIF_DGI)
    ni_gratif_imposable_ir   = MAX(0, ni_gratif_sociale - PLAFOND_NI_GRATIF_DGI)
    ni_gratif_imposable_cnss = MAX(0, ni_gratif_sociale - PLAFOND_NI_GRATIF_ANNUEL)
    NOTE : "Plafonds annuels — cumuler les versements de l'année"
  }

  RULE ni.12 — Autres NI {
    ni_autres_exo       = ni_autres
    ni_autres_imposable = 0
  }

  RULE ni.13 — Agrégation {
    total_ni_exonere = ni_transport_exo + ni_kilometrique_exo + ni_tournee_exo
                     + ni_representation_exo + ni_panier_exo
                     + ni_caisse_exo_dgi + ni_lait_exo_dgi
                     + ni_outillage_exo_dgi + ni_salissure_exo_dgi
                     + ni_aide_medicale_exo + ni_gratif_exo_dgi + ni_autres_exo

    total_ni_excedent_imposable = ni_transport_imposable + ni_tournee_imposable
                                + ni_representation_imposable + ni_panier_imposable
                                + ni_caisse_imposable_ir + ni_lait_imposable_ir
                                + ni_outillage_imposable_ir + ni_salissure_imposable_ir
                                + ni_gratif_imposable_ir
  }
}


MODULE[05] salaire_brut_imposable {
  INPUTS  : salaire_base_mensuel, prime_anciennete, total_hsupp,
            primes_imposables,
            total_ni_excedent_imposable
  OUTPUTS : salaire_brut_imposable, total_primes_imposables

  RULE sbi.1 {
    ;; Calculer la somme de TOUTES les primes imposables
    total_primes_imposables = SUM(primes_imposables[*].montant)
    
    salaire_brut_imposable = salaire_base_mensuel
                           + prime_anciennete
                           + total_hsupp
                           + total_primes_imposables
                           + total_ni_excedent_imposable
  }
}


MODULE[06] cnss {
  INPUTS  : salaire_brut_imposable
  OUTPUTS : base_cnss_rg,
            cnss_rg_salarial, cnss_amo_salarial, total_cnss_salarial,
            cnss_rg_patronal, cnss_alloc_fam_patronal, cnss_fp_patronal,
            cnss_amo_patronal, cnss_particip_amo_patronal, total_cnss_patronal

  RULE cnss.1 — Base plafonnée RG {
    base_cnss_rg = MIN(salaire_brut_imposable, PLAFOND_CNSS_MENSUEL)
    ;; Le Régime Général est PLAFONNÉ à 6 000 MAD/mois
    ;; L'AMO est NON PLAFONNÉE (s'applique sur brut imposable total)
  }

  RULE cnss.2 — Part salariale {
    cnss_rg_salarial    = ROUND(base_cnss_rg              * CNSS_RG_SALARIAL,   2)
    cnss_amo_salarial   = ROUND(salaire_brut_imposable     * CNSS_AMO_SALARIAL,  2)
    total_cnss_salarial = cnss_rg_salarial + cnss_amo_salarial
  }

  RULE cnss.3 — Part patronale {
    cnss_rg_patronal           = ROUND(base_cnss_rg          * CNSS_RG_PATRONAL,          2)
    cnss_alloc_fam_patronal    = ROUND(salaire_brut_imposable * CNSS_ALLOC_FAM_PAT,        2)
    cnss_fp_patronal           = ROUND(salaire_brut_imposable * CNSS_FP_PAT,               2)
    cnss_amo_patronal          = ROUND(salaire_brut_imposable * CNSS_AMO_PATRONAL,         2)
    cnss_particip_amo_patronal = ROUND(salaire_brut_imposable * CNSS_AMO_PARTICIPATION_PAT,2)
    total_cnss_patronal = cnss_rg_patronal + cnss_alloc_fam_patronal
                        + cnss_fp_patronal + cnss_amo_patronal
                        + cnss_particip_amo_patronal
  }

  TABLE cnss.taux {
    | branche                   | salariale | patronale | plafond        |
    |---------------------------|-----------|-----------|----------------|
    | Prestations sociales (RG) |  4.48%    |  8.98%    | 6 000 MAD/mois |
    | Allocations familiales    |  0%       |  6.40%    | Aucun          |
    | Formation professionnelle |  0%       |  1.60%    | Aucun          |
    | AMO cotisation            |  2.26%    |  2.26%    | Aucun          |
    | AMO participation         |  0%       |  1.85%    | Aucun          |
    | TOTAL                     |  6.74%    | 21.09%    |                |
  }
}


MODULE[07] cimr {
  INPUTS  : salaire_brut_imposable, regime_cimr,
            cimr_taux_salarial, cimr_taux_patronal
  OUTPUTS : cimr_salarial, cimr_patronal

  RULE cimr.1 {
    WHEN regime_cimr = AUCUN {
      cimr_salarial = 0
      cimr_patronal = 0
    }
    WHEN regime_cimr = AL_KAMIL {
      cimr_salarial = ROUND(salaire_brut_imposable * cimr_taux_salarial, 2)
      cimr_patronal = ROUND(salaire_brut_imposable * cimr_taux_patronal, 2)
    }
    WHEN regime_cimr = AL_MOUNASSIB {
      base_cimr_mounassib = MAX(0, salaire_brut_imposable - PLAFOND_CNSS_MENSUEL)
      cimr_salarial = ROUND(base_cimr_mounassib * cimr_taux_salarial, 2)
      cimr_patronal = ROUND(base_cimr_mounassib * cimr_taux_patronal, 2)
    }
  }
}


;;; ============================================================
;;; ⚠️  MODULE[08] — FRAIS PROFESSIONNELS
;;; ATTENTION : Ce module calcule UNIQUEMENT le montant de l'abattement FP.
;;;             Il ne calcule PAS le revenu net imposable.
;;;             Le RNI est calculé dans MODULE[09] en soustrayant
;;;             AUSSI les cotisations salariales (cnss + cimr + mutuelle).
;;; ============================================================

MODULE[08] frais_professionnels {
  INPUTS  : salaire_brut_imposable
  OUTPUTS : base_fp, taux_fp, plafond_fp, montant_fp

  RULE fp.1 — Base de calcul du taux FP {
    ;;
    ;; ╔══════════════════════════════════════════════════════════╗
    ;; ║  base_fp = salaire_brut_imposable                       ║
    ;; ║                                                          ║
    ;; ║  ❌ ERREUR FRÉQUENTE : soustraire CNSS avant le calcul  ║
    ;; ║     du taux. Le % FP s'applique sur le BRUT IMPOSABLE   ║
    ;; ║     TOTAL, avant toute autre déduction.                  ║
    ;; ║                                                          ║
    ;; ║  ✅ CORRECT :                                            ║
    ;; ║     base_fp = salaire_brut_imposable  (brut total)       ║
    ;; ║     montant_fp = MIN(base_fp × taux, plafond)            ║
    ;; ║                                                          ║
    ;; ║  Le montant_fp sera déduit dans MODULE[09]               ║
    ;; ║  CONJOINTEMENT avec CNSS, CIMR, Mutuelle.               ║
    ;; ╚══════════════════════════════════════════════════════════╝
    ;;
    base_fp = salaire_brut_imposable
  }

  RULE fp.2 — Taux et plafond {
    WHEN base_fp <= 6500.00 {
      taux_fp    = 0.35
      plafond_fp = 2916.67
    }
    WHEN base_fp > 6500.00 {
      taux_fp    = 0.25
      plafond_fp = 2916.67
    }
    ;;
    ;; ⚠️  ATTENTION IA : le plafond est un MAXIMUM, pas un forfait.
    ;; Exemple : base_fp = 9 900 → 9 900 × 25% = 2 475 < 2916.67 → fp = 2 475
  }

  RULE fp.3 — Montant déductible {
    montant_fp = MIN(ROUND(base_fp * taux_fp, 2), plafond_fp)
  }

  TABLE fp.seuil_plafonnement {
    ;; Seuil à partir duquel le plafond est actif
    | taux | plafond   | brut_min_pour_plafond |
    |------|-----------|-----------------------|
    | 35%  | 2 916.67  | --------- MAD          |
    | 25%  | 2 916,67  | 11 666,68 MAD         |
  }
}


;;; ============================================================
;;; ⚠️  MODULE[09] — BASE IR (REVENU NET IMPOSABLE)
;;; C'EST ICI QUE TOUTES LES DÉDUCTIONS S'ACCUMULENT avant IR.
;;; ============================================================

MODULE[09] base_ir {
  INPUTS  : salaire_brut_imposable,
            total_cnss_salarial,
            cimr_salarial,
            mutuelle_salariale,
            montant_fp,
            interet_pret_logement
  OUTPUTS : revenu_net_imposable

  RULE ir_base.1 — Revenu Net Imposable {
    ;;
    ;; ╔══════════════════════════════════════════════════════════════╗
    ;; ║  FORMULE COMPLÈTE ET OBLIGATOIRE :                          ║
    ;; ║                                                              ║
    ;; ║  RNI = salaire_brut_imposable                               ║
    ;; ║        − total_cnss_salarial      ← CNSS RG + AMO          ║
    ;; ║        − cimr_salarial            ← si applicable           ║
    ;; ║        − mutuelle_salariale       ← si applicable           ║
    ;; ║        − montant_fp               ← abattement FP           ║
    ;; ║        − interet_pret_logement    ← si applicable           ║
    ;; ║                                                              ║
    ;; ║  ❌ ERREUR FRÉQUENTE : oublier total_cnss_salarial          ║
    ;; ║     ou cimr_salarial dans la déduction.                     ║
    ;; ║                                                              ║
    ;; ║  VÉRIFICATION : RNI doit être INFÉRIEUR à                   ║
    ;; ║  (brut_imposable − montant_fp) car on déduit aussi CNSS.    ║
    ;; ╚══════════════════════════════════════════════════════════════╝
    ;;
    revenu_net_imposable = salaire_brut_imposable
                         - total_cnss_salarial
                         - cimr_salarial
                         - mutuelle_salariale
                         - montant_fp
                         - interet_pret_logement

    revenu_net_imposable = MAX(0, revenu_net_imposable)
  }

  SELF_CHECK {
    ;; L'IA DOIT vérifier cette inégalité avant de continuer :
    ASSERT revenu_net_imposable < (salaire_brut_imposable - montant_fp)
      OR   total_cnss_salarial = 0
      MESSAGE "⚠️ RNI suspects : les cotisations salariales n'ont pas été déduites !"
  }
}


MODULE[10] ir {
  INPUTS  : revenu_net_imposable, situation_fam
  OUTPUTS : taux_ir, deduction_bareme, ir_brut, deduction_famille, ir_final

  RULE ir.1 — Barème mensuel 2026 {
    WHEN revenu_net_imposable <= 3333.33  { taux_ir = 0.00  deduction_bareme =    0.00 }
    WHEN revenu_net_imposable <= 5000.00  { taux_ir = 0.10  deduction_bareme =  333.33 }
    WHEN revenu_net_imposable <= 6666.67  { taux_ir = 0.20  deduction_bareme =  833.33 }
    WHEN revenu_net_imposable <= 8333.33  { taux_ir = 0.30  deduction_bareme = 1500.00 }
    WHEN revenu_net_imposable <= 15000.00 { taux_ir = 0.34  deduction_bareme = 1833.33 }
    WHEN revenu_net_imposable >  15000.00 { taux_ir = 0.37  deduction_bareme = 2283.33 }
    ir_brut = ROUND(revenu_net_imposable * taux_ir, 2)
  }

  RULE ir.2 — Charges de famille {
    deduction_famille = situation_fam * IR_DEDUCTION_FAMILLE
    ;; 30 MAD/mois par personne à charge (conjoint + enfants)
  }

  RULE ir.3 — IR final {
    ir_final = MAX(0, ROUND(ir_brut - deduction_bareme - deduction_famille, 2))
  }

  TABLE ir.bareme_mensuel {
    | RNI_min (MAD)  | RNI_max (MAD)  | taux  | déduction mensuelle |
    |----------------|----------------|-------|---------------------|
    | 0              | 3 333,33       |  0%   |       0,00          |
    | 3 333,34       | 5 000,00       | 10%   |     333,33          |
    | 5 000,01       | 6 666,67       | 20%   |     833,33          |
    | 6 666,68       | 8 333,33       | 30%   |   1 500,00          |
    | 8 333,34       | 15 000,00      | 34%   |   1 833,33          |
    | 15 000,01      | ∞              | 37%   |   2 283,33          |
  }
}

;;; SECTION 8 — ERREURS COMMUNES À ÉVITER
;;; (anti-patterns pour IA)

@ANTIPATTERNS {

  ANTIPATTERN ap.1 — FP calculé sur base réduite {
    ;;
    ;; ❌ FAUX :
    ;;    base_fp = salaire_brut_imposable − total_cnss_salarial
    ;;    montant_fp = base_fp × 25%
    ;;
    ;; ✅ CORRECT :
    ;;    base_fp = salaire_brut_imposable
    ;;    montant_fp = MIN(base_fp × 25%, 2500)
    ;;    → Les cotisations CNSS sont déduites dans MODULE[09], pas ici.
  }

  ANTIPATTERN ap.2 — Plafond FP comme forfait {
    ;;
    ;; ❌ FAUX :
    ;;    montant_fp = 2500  (pour tout brut > 6500)
    ;;
    ;; ✅ CORRECT :
    ;;    montant_fp = MIN(brut × taux, plafond)
    ;;    Si brut = 9900 → MIN(2475, 2500) = 2475  ← pas 2500
    ;;    Si brut = 12000 → MIN(3000, 2500) = 2500  ← plafond actif
  }

  ANTIPATTERN ap.3 — RNI sans déduction CNSS {
    ;;
    ;; ❌ FAUX :
    ;;    RNI = brut_imposable − montant_fp
    ;;
    ;; ✅ CORRECT :
    ;;    RNI = brut_imposable − cnss_salarial − cimr_salarial
    ;;                         − mutuelle_salariale − montant_fp
    ;;                         − interet_pret_logement
  }

  ANTIPATTERN ap.4 — Ancienneté arrondie à l'excès {
    ;;
    ;; ❌ FAUX :
    ;;    5 ans et 3 mois → taux 10%  ✅ correct
    ;;    4 ans et 11 mois → taux 10%  ❌ FAUX
    ;;
    ;; ✅ CORRECT :
    ;;    Compter les années ENTIÈRES révolues (DIFF_YEARS floor)
    ;;    4 ans 11 mois → anciennete_annees = 4 → taux 5%
  }

  ANTIPATTERN ap.5 — CNSS RG non plafonnée {
    ;;
    ;; ❌ FAUX :
    ;;    cnss_rg_salarial = salaire_brut_imposable × 4.48%
    ;;
    ;; ✅ CORRECT :
    ;;    base_cnss_rg = MIN(salaire_brut_imposable, 6000)
    ;;    cnss_rg_salarial = base_cnss_rg × 4.48%
    ;;    (l'AMO seule n'est pas plafonnée)
  }
